let mytable = new Vue({
    el: '#mytable',
    data() {
        return {
            datalist: {}
        }
    },
    created: function() {
        let that = this;
        $.ajax({
            type: 'get',
            url: "http://cnodejs.org/api/v1/topics",
            dataType: 'json',
            success: function(list) {
                if (list.success) {
                    that.datalist = list.data;
                } else {
                    alert("接口调用失败");
                }
            },
            error: function(list) {
                alert(JSON.stringify(list));
            }
        })
    }
})
