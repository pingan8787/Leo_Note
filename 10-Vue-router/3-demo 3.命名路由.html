                    <!--要使用v-bind绑定to-->

const UserId = {
    template: '<div>这是我的命名路由，名字叫{{$route.name}}，参数叫{{$route.params.userId}}</div>'
}
const router4 = new VueRouter({
    routes: [{
        path: '/user/:userId',
        name: 'user',
        component: UserId
    }]
})
const app4 = new Vue({
    router: router4
}).$mount('#mytest4');
