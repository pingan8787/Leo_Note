angular操作Calss操作
  1. <!DOCTYPE html>  
  2. <html><head>  
  3.  <meta charset="UTF-8">  
  4.         <script src="js/angular.min.js"></script>  
  5.         <style type="text/css">  
  6.             .red{width: 100px; height: 100px; background: red;  }  
  7.             .green{width: 100px; height: 100px;background: green;}  
  8.         </style>  
  9.     </head>  
  10.     <body>  
  11.         <div ng-app="my-app">  
  12.             <div ng-controller="frist">  
  13.                 <!--第一种  直接加变量-->  
  14.                 <div ng-class="tempClass"></div>  
  15.                 <!--第二种  用{{}} 包住的变量表达式-->  
  16.                 <div ng-class="{{bClass}}"></div>  
  17.                 <!--第三种-->  
  18.                 <div ng-class="{'red':num,'green':!num}"></div>  
  19.                 <!--第四种-->  
  20.                 <div ng-class="{true:'green',false:'red'}[num]"></div>  
  21.                 <!--第五种-->  
  22.                            写在标签内 就不用{{}}
  23.                <span>
  24.                    <span ng-repeat="(a,b) in value.bid_number track by $index" ng-class="'bid_number'+(b | numberfilter:a:b)">{{b | numberfilter:a:b}}</span>
  25.                </span>
  26.             </div>  
  27.         </div> 
  28.         <script type="text/javascript">  
  29.             var app = angular.module('my-app',[]);  
  30.             app.controller('frist',function($scope){  
  31.                     $scope.tempClass = 'green';  
  32.                     $scope.bClass = "{'red':true,'green':false}"  
  33.                     $scope.num = true;        
  34.             })  
  35.         </script>  
  36.     </body>
