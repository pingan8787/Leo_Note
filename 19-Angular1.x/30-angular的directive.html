<div id="left_nav" ng-controller="left_nav_controller">
<a href="#/manager_login" class="manager_login"><span class="glyphicon glyphicon-user"></span> 管理员登录</a>
<div ng-repeat="list in left_list" tooglenav>
<h2><span class="glyphicon {{list.titleIcon}}"></span> {{list.titleName}}</h2>
<ul class="hidetabs">
<li ng-repeat="li in list.branch"><a ng-href="#/{{li.branchUrl}}"><span class="glyphicon glyphicon-chevron-right"></span> {{li.branchName}}</a></li>
</ul>
</div>
</div>

app.directive('tooglenav', function() {
return {
restrict: 'A',
link: function(scope, elem, attr) {
elem.find('h2').on('click', function() {
if (elem.find('ul').hasClass('showtabs')) {
elem.find('ul').removeClass('showtabs');
} else {
elem.find('ul').addClass('showtabs');
}
})
}
}
})
