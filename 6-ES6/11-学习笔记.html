/** 数组 **/

/** 函数 **/
(() => { console.log('Hello!') })();
//bad
[1, 2, 3].map(function(x) {
return x * x;
})

[1, 2, 3].map((x) => {
return x * x;
})

/** Map结构 **/
let arr = [1, 2, 3, 4];
let map = new Map(arr);
for (let key of map.keys()) {
console.log(key);
}
for (let value of map.values()) {
console.log(value);
}
for (let item of map.entries()) {
console.log(item[0] + ',' + item[1]);
}
